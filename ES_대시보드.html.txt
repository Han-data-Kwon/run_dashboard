function parseCSV(buf){
  var text;
  try{text=new TextDecoder('euc-kr').decode(buf);}catch(e){text=new TextDecoder('utf-8').decode(buf);}
  if(!text.includes('상태')){try{text=new TextDecoder('utf-8').decode(buf);}catch(e){}}
  var lines=text.split(/\r?\n/).filter(function(l){return l.trim();});
  if(lines.length<2)return[];
  var hdrs=spL(lines[0]),hm={};
  hdrs.forEach(function(h,i){hm[h.trim()]=i;});
  var cn=['예약번호','상태','결제일자일','취소일자일','체크인일자일','체크아웃일자일','회원번호','가입일자','직전구매일','최초구매일','기업번호','기업명','제휴점번호','제휴점명','대지역','소지역','카테고리1','카테고리2','업무사용유무','유상사용유무','숙박일수','gmv','총포인트금액','유상포인트','환불유상포인트','유상쿠폰','기본쿠폰번호','기본쿠폰명','기본쿠폰금액','중복쿠폰번호','중복쿠폰명','중복쿠폰금액'];
  var im=cn.map(function(n){if(hm[n]!==undefined)return hm[n];var k=Object.keys(hm).find(function(k){return k.includes(n)||n.includes(k);});return k!==undefined?hm[k]:-1;});
  return lines.slice(1).map(function(l){var f=spL(l);return im.map(function(i){return i>=0?(f[i]||''):'';});});
}
function spL(line){
  var res=[],s=0,q=false;
  for(var i=0;i<line.length;i++){if(line[i]==='"'){q=!q;continue;}if(line[i]===','&&!q){res.push(line.slice(s,i).replace(/"/g,'').trim());s=i+1;}}
  res.push(line.slice(s).replace(/"/g,'').trim());return res;
}
